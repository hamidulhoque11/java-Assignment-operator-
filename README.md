# java-Assignment-operator-

/*
Assignment operator -
    1. +=
    2. -=
    3. *=
    4. /=
    5. %=
    
-------------------------------------------- */
import java.util.Scanner;
class Main {
   public static void main (String [] args){
       
       //  applying : +=
     int x=3;
     int y=2;
     x += y;
     System.out.println("x += y or x = x+y :"+x);
     
       //  applying : -=
      x=3;
      y=2;
     x -= y;
     System.out.println("x -= y or x = x-y :"+x);
      
       //  applying : *=
     x=3;
      y=2;
     x *= y;
     System.out.println("x *= y or x = x*y :"+x);
     
     //applying : /=
      x=3;
      y=2;
     x /= y;
     System.out.println("x /= y or x = x/y :"+x);
     
     //applying : %=
      x=3;
      y=2;
     x %= y;
     System.out.println("x %= y or x = x%y :"+x);
    }
}
